414 template <typename internal_int_t, std::size_t l_Q, std::size_t r_Q>
415 auto fixed_mul(fixed_t<internal_int_t, l_Q> lhs,
416  const fixed_t<internal_int_t, r_Q> rhs)
417  -> fixed_t<impl::wider_int_t<internal_int_t>, l_Q + r_Q> {
418  using wider_int_t = impl::wider_int_t<internal_int_t>;
419  const auto lhs_wide = static_cast<wider_int_t>(lhs.fixed_point_);
420  const auto rhs_wide = static_cast<wider_int_t>(rhs.fixed_point_);
421 
422  using result_t = fixed_t<wider_int_t, l_Q + r_Q>;
423 
424  return result_t(lhs_wide * rhs_wide, true);
425 }

442 template <typename internal_int_t, std::size_t l_Q, std::size_t r_Q>
443 auto fixed_div(fixed_t<internal_int_t, l_Q> lhs,
444  const fixed_t<internal_int_t, r_Q> rhs)
445  -> fixed_t<impl::wider_int_t<internal_int_t>,
446  l_Q
447  + (sizeof(impl::wider_int_t<internal_int_t>) * 8U
448  - sizeof(internal_int_t) * 8U)
449  - r_Q> {
450  using wider_int_t = impl::wider_int_t<internal_int_t>;
451  constexpr std::size_t kResultBits = sizeof(wider_int_t) * 8U;
452  constexpr std::size_t kSrcBits = sizeof(internal_int_t) * 8U;
453  const auto lhs_wide = static_cast<wider_int_t>(lhs.fixed_point_);
454  const auto rhs_wide = static_cast<wider_int_t>(rhs.fixed_point_);
455 
456  using result_t = fixed_t<wider_int_t, l_Q + (kResultBits - kSrcBits) - r_Q>;
457 
458  return result_t((lhs_wide << (kResultBits - kSrcBits)) / rhs_wide, true);
459 }
