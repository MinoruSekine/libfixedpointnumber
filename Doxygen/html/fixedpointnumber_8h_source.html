417 template <typename internal_int_t, std::size_t l_Q, std::size_t r_Q>
418 auto fixed_mul(fixed_t<internal_int_t, l_Q> lhs,
419  const fixed_t<internal_int_t, r_Q> rhs)
420  -> fixed_t<impl::wider_int_t<internal_int_t>, l_Q + r_Q> {
421  using wider_int_t = impl::wider_int_t<internal_int_t>;
422  const auto lhs_wide = static_cast<wider_int_t>(lhs.fixed_point_);
423  const auto rhs_wide = static_cast<wider_int_t>(rhs.fixed_point_);
424 
425  using result_t = fixed_t<wider_int_t, l_Q + r_Q>;
426 
427  return result_t(static_cast<wider_int_t>(lhs_wide * rhs_wide), true);
428 }

445 template <typename internal_int_t, std::size_t l_Q, std::size_t r_Q>
446 auto fixed_div(fixed_t<internal_int_t, l_Q> lhs,
447  const fixed_t<internal_int_t, r_Q> rhs)
448  -> fixed_t<impl::wider_int_t<internal_int_t>,
449  l_Q
450  + (sizeof(impl::wider_int_t<internal_int_t>) * 8U
451  - sizeof(internal_int_t) * 8U)
452  - r_Q> {
453  using wider_int_t = impl::wider_int_t<internal_int_t>;
454  constexpr std::size_t kResultBits = sizeof(wider_int_t) * 8U;
455  constexpr std::size_t kSrcBits = sizeof(internal_int_t) * 8U;
456  const auto lhs_wide = static_cast<wider_int_t>(lhs.fixed_point_);
457  const auto rhs_wide = static_cast<wider_int_t>(rhs.fixed_point_);
458 
459  using result_t = fixed_t<wider_int_t, l_Q + (kResultBits - kSrcBits) - r_Q>;
460 
461  const auto wider_int_result =
462  static_cast<wider_int_t>((lhs_wide << (kResultBits - kSrcBits))
463  / rhs_wide);
464 
465  return result_t(wider_int_result, true);
466 }
