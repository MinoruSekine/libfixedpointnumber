426 template <typename internal_int_t, std::size_t l_Q, std::size_t r_Q>
427 auto fixed_mul(fixed_t<internal_int_t, l_Q> lhs,
428  const fixed_t<internal_int_t, r_Q> rhs)
429  -> fixed_t<impl::wider_int_t<internal_int_t>, l_Q + r_Q> {
430  using wider_int_t = impl::wider_int_t<internal_int_t>;
431  const auto lhs_wide = static_cast<wider_int_t>(lhs.fixed_point_);
432  const auto rhs_wide = static_cast<wider_int_t>(rhs.fixed_point_);
433 
434  using result_t = fixed_t<wider_int_t, l_Q + r_Q>;
435 
436  return result_t(static_cast<wider_int_t>(lhs_wide * rhs_wide), true);
437 }

454 template <typename internal_int_t, std::size_t l_Q, std::size_t r_Q>
455 auto fixed_div(fixed_t<internal_int_t, l_Q> lhs,
456  const fixed_t<internal_int_t, r_Q> rhs)
457  -> fixed_t<impl::wider_int_t<internal_int_t>,
458  l_Q
459  + (sizeof(impl::wider_int_t<internal_int_t>) * 8U
460  - sizeof(internal_int_t) * 8U)
461  - r_Q> {
462  using wider_int_t = impl::wider_int_t<internal_int_t>;
463  constexpr std::size_t kResultBits = sizeof(wider_int_t) * 8U;
464  constexpr std::size_t kSrcBits = sizeof(internal_int_t) * 8U;
465  const auto lhs_wide = static_cast<wider_int_t>(lhs.fixed_point_);
466  const auto rhs_wide = static_cast<wider_int_t>(rhs.fixed_point_);
467 
468  using result_t = fixed_t<wider_int_t, l_Q + (kResultBits - kSrcBits) - r_Q>;
469 
470  const auto wider_int_result =
471  static_cast<wider_int_t>((lhs_wide << (kResultBits - kSrcBits))
472  / rhs_wide);
473 
474  return result_t(wider_int_result, true);
475 }
