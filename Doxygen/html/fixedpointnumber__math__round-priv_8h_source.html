32 template <typename IntType, std::size_t Q>
33 constexpr fixed_t<IntType, Q> fixed_floor_positive(fixed_t<IntType, Q> src) {
34  return fixed_t<IntType, Q>(src.fixed_point_ & src.kIntegralPartMask, true);
35 }
36 
37 template <typename IntType, std::size_t Q>
38 constexpr fixed_t<IntType, Q> fixed_ceil_positive(fixed_t<IntType, Q> src) {
39  return (((src.fixed_point_ & src.kDecimalPartMask) != 0)
40  ? (fixed_floor_positive(src) + fixed_t<IntType, Q>(1))
41  : src);
42 }

44 template <typename IntType, std::size_t Q>
45 constexpr fixed_t<IntType, Q> GetZeroPointFive() {
46  // From string ctor like as fixed_t<IntType, Q>("0.5") can't be suitable here,
47  // because that ctor is not constexpr ctor now.
48  return fixed_t<IntType, Q>(static_cast<IntType>(1 << (Q - 1)), true);
49 }

51 template <typename IntType, std::size_t Q>
52 constexpr fixed_t<IntType, Q> fixed_round_positive(fixed_t<IntType, Q> src) {
53  return fixed_floor_positive(src + GetZeroPointFive<IntType, Q>());
54 }

56 template <typename IntType, std::size_t Q>
57 constexpr fixed_t<IntType, Q> fixed_floor_negative(fixed_t<IntType, Q> src) {
58  return -fixed_ceil_positive(-src);
59 }
60 
61 template <typename IntType, std::size_t Q>
62 constexpr fixed_t<IntType, Q> fixed_ceil_negative(fixed_t<IntType, Q> src) {
63  return -fixed_floor_positive(-src);
64 }
65 
66 template <typename IntType, std::size_t Q>
67 constexpr fixed_t<IntType, Q> fixed_round_negative(fixed_t<IntType, Q> src) {
68  return fixed_ceil_negative(src - GetZeroPointFive<IntType, Q>());
69 }
